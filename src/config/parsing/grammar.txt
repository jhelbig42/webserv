config      -> *sep expression *(*sep expression)

newline     -> "\n"
ws          -> +(" " | "\t")

expression  -> server
            -> general

general     -> backlog
            -> max_request_size

server      -> "server" *sep "{" *sep entries *sep "}"

entries     -> entry *(*sep entry)

entry       -> (listen | root | allowed | redirect | index | directory_listing | max_request_size | backlog) *sep ";"

listen      -> "listen" +ws listen-val
listen-val  -> (ip | "*") ":" (port | "*")
ip          -> ipv4 | ipv6
ipv4        -> uint8 "." uint8 "." uint8 "." uint8 
ipv6        -> DUMMY
port        -> uint16

root        -> "root" +ws root-val
root-val    -> abspath
abspath     -> "/" *(+pathchar "/")

allowed     -> "allowed" +ws allowed-val
allowed-val -> "*" | method *(+ws method)
method      -> "POST" | "GET" | "PUT" | "DELETE" | "HEAD"

index       -> "index" +ws name *(+ws name)

directory_listing
            -> "directory_listing" +ws "true" | "false"

max_request_size
            -> "max_request_size" +ws off_t

backlog     -> "backlog" +ws int-non-negative

redirect    -> "redirect" +ws redirect-scope *sep "{" *sep redirect-rules *sep +newline *sep "}"

redirect-scope
            -> abs-path

content-type
            -> FROM_HTTP

redirect-rules
            -> redirect-rule *(*sep redirect-rule)

sep         -> newline | ws

redirect-rule
            -> (redirect | execute-rule | path-rule | resource-rule | code-rule) *sep ";"

execute-rule
            -> content-type +ws executor

path-rule   -> abs-path

resource-rule
            -> resource

code-rule   -> status-code +ws resource

status-code -> FROM_HTTP

resource    -> ?abs-path name
